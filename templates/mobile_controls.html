}) .catch(error => { console.error('Error stopping player:', error); }); } function nextSong() { fetch('/api/player/next', { method: 'POST' }) .then(response => response.json()) .then(data => { console.log('Next song loaded'); }) .catch(error => { console.error('Error loading next song:', error); }); } function playPause() { fetch('/api/player/play_pause', { method: 'POST' }) .then(response => response.json()) .then(data => { console.log('Play/pause toggled'); }) .catch(error => { console.error('Error toggling play/pause:', error); }); } function toggleFullscreen() { fetch('/api/player/toggle_fullscreen', { method: 'POST' }) .then(response => response.json()) .then(data => { console.log('Fullscreen toggled'); }) .catch(error => { console.error('Error toggling fullscreen:', error); }); } function restartSong() { fetch('/api/player/restart', { method: 'POST' }) .then(response => response.json()) .then(data => { console.log('Song restarted'); }) .catch(error => { console.error('Error restarting song:', error); }); } function toggleMute() { fetch('/api/player/toggle_mute', { method: 'POST' }) .then(response => response.json()) .then(data => { console.log('Mute toggled'); }) .catch(error => { console.error('Error toggling mute:', error); }); } // Function to update current song display function updateCurrentSong(song) { console.log('Updating current song:', song); const noCurrentSongEl = document.getElementById('no-current-song'); const currentSongInfoEl = document.getElementById('current-song-info'); const currentTitleEl = document.getElementById('current-title'); const currentSingerEl = document.getElementById('current-singer'); if (!noCurrentSongEl || !currentSongInfoEl || !currentTitleEl || !currentSingerEl) { console.error('Missing elements for current song display'); return; } if (song) { console.log('Setting current song title to:', song.title); console.log('Setting current song singer to:', song.singer); currentTitleEl.textContent = song.title; currentSingerEl.textContent = song.singer; noCurrentSongEl.classList.add('d-none'); currentSongInfoEl.classList.remove('d-none'); // Enable restart button if player is running const restartBtn = document.getElementById('restart-song'); if (restartBtn && !restartBtn.disabled) { restartBtn.disabled = false; } } else { console.log('No current song to display'); noCurrentSongEl.classList.remove('d-none'); currentSongInfoEl.classList.add('d-none'); // Disable restart button const restartBtn = document.getElementById('restart-song'); if (restartBtn) { restartBtn.disabled = true; } } } // Function to update next song display function updateNextSong(nextSong) { console.log('Updating next song:', nextSong); const noNextSongEl = document.getElementById('no-next-song'); const nextSongInfoEl = document.getElementById('next-song-info'); const nextTitleEl = document.getElementById('next-title'); const nextSingerEl = document.getElementById('next-singer'); if (!noNextSongEl || !nextSongInfoEl || !nextTitleEl || !nextSingerEl) { console.error('Missing elements for next song display'); return; } if (nextSong) { console.log('Setting next song title to:', nextSong.title); console.log('Setting next song singer to:', nextSong.singer); nextTitleEl.textContent = nextSong.title; nextSingerEl.textContent = nextSong.singer; noNextSongEl.classList.add('d-none'); nextSongInfoEl.classList.remove('d-none'); } else { console.log('No next song to display'); noNextSongEl.classList.remove('d-none'); nextSongInfoEl.classList.add('d-none'); } } // Initialize current song and queue on page load document.addEventListener('DOMContentLoaded', function() { console.log('Initializing mobile controls page'); // Check player status and get queue data fetch('/api/player/status') .then(response => response.json()) .then(data => { console.log('Player status:', data); // Get queue data to update both current and next song fetch('/api/queue') .then(response => response.json()) .then(queueData => { console.log('Queue data:', queueData); // Update current song if available if (data.has_current_song && queueData.current_song) { console.log('Current song available, updating display'); updateCurrentSong(queueData.current_song); } else { console.log('No current song available'); updateCurrentSong(null); } // Update next song if available if (queueData.queue && queueData.queue.length > 0) { console.log('Next song available, updating display'); updateNextSong(queueData.queue[0]); } else { console.log('No next song available'); updateNextSong(null); } }) .catch(error => { console.error('Error fetching queue data:', error); }); }); // Set up socket listener for queue updates socket.on('queue_update', function(data) { if (data.queue && data.queue.length > 0) { updateNextSong(data.queue[0]); } else { updateNextSong(null); } }); // Set up socket listener for current song updates socket.on('current_song_update', function(data) { updateCurrentSong(data.current_song); // When current song changes, refresh the queue to update the next song fetch('/api/queue') .then(response => response.json()) .then(queueData => { if (queueData.queue && queueData.queue.length > 0) { updateNextSong(queueData.queue[0]); } else { updateNextSong(null); } }); }); }); {% endblock %}
